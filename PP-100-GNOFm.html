import ipywidgets as widgets
from IPython.display import display

# Create the tabs
tab_contents = [
    widgets.HTML(value="<h1>Tab 1 Content</h1><p>This is the content of the first tab.</p>"),
    widgets.HTML(value="<h1>Tab 2 Content</h1><p>This is the content of the second tab.</p>"),
    widgets.HTML(value="<h1>Tab 3 Content</h1><p>This is the content of the third tab.</p>")
]

tab = widgets.Tab(children=tab_contents)

# Create the dropdown menu
dropdown_options = {
    "Tab 1": 0,
    "Tab 2": 1,
    "Tab 3": 2
}

dropdown = widgets.Dropdown(
    options=dropdown_options,
    value=0,
    description='Select Tab:'
)

# Link the dropdown to the tab using jslink
widgets.jslink((dropdown, 'index'), (tab, 'selected_index'))

# Display the widgets
display(dropdown)
display(tab)
